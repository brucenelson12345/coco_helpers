import onnx
import numpy as np
from aimet_onnx.quantsim import QuantizationSimModel

# 1. Load your patched ONNX model
model = onnx.load("blackjeep_patched.onnx")

# 2. Define a forward pass function for AIMET to trace the graph
def forward_pass(session, _):
    # AIMET will use this to push data through. 
    # For accurate scales, replace this dummy tensor with a loop over a batch of your actual real images!
    dummy_input = np.random.rand(1, 3, 384, 384).astype(np.float32)
    input_name = session.get_inputs()[0].name
    session.run(None, {input_name: dummy_input})

# 3. Initialize the Quantization Simulator
sim = QuantizationSimModel(
    model, 
    dummy_input=np.random.rand(1, 3, 384, 384).astype(np.float32), 
    default_param_bw=8, 
    default_activation_bw=8,
    use_cuda=True # Set to False if you aren't passing the GPU through WSL
)

# 4. Compute the encodings (this runs your forward pass)
sim.compute_encodings(forward_pass, forward_pass_callback_args=None)

# 5. Export the mixed-precision encodings
sim.export("./aimet_output", "blackjeep_mixed")
