import cv2
import numpy as np
from typing import Tuple

def draw_segmentation_mask(
    frame: np.ndarray,
    mask: np.ndarray,
    color: Tuple[int, int, int] = (0, 140, 255), # BGR for Orange
    outline_color: Tuple[int, int, int] = (255, 255, 255), # BGR for White
    text: str = None,
    alpha: float = 0.6
):
    """
    Draws a semi-transparent segmentation mask with a solid outline and text overlay.
    """
    # 1. Create and blend the semi-transparent mask
    colored_overlay = np.zeros_like(frame, dtype=np.uint8)
    colored_overlay[mask > 0] = color
    cv2.addWeighted(colored_overlay, alpha, frame, 1.0, 0, frame)
    
    # 2. Extract contours to draw the white outline
    mask_uint8 = (mask * 255).astype(np.uint8)
    contours, _ = cv2.findContours(mask_uint8, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    
    # Draw the white outline with a thickness of 2
    cv2.drawContours(frame, contours, -1, outline_color, thickness=2)
    
    # 3. Add text overlay if provided
    if text is not None and len(contours) > 0:
        # Find the largest contour to center the text appropriately
        largest_contour = max(contours, key=cv2.contourArea)
        x, y, w, h = cv2.boundingRect(largest_contour)
        
        # Styling for text and its background
        bg_color = (0, 100, 230)  # Darker orange background for text
        text_color = (0, 0, 0)    # Black text
        font = cv2.FONT_HERSHEY_SIMPLEX
        font_scale = 0.7
        thickness = 1
        
        # Calculate text size and placement (centered across the bounding box of the mask)
        (text_width, text_height), _ = cv2.getTextSize(text, font, font_scale, thickness)
        text_x = x + (w - text_width) // 2
        text_y = y + (h + text_height) // 2
        
        # Draw the solid dark orange text background
        padding_x = 8
        padding_y = 6
        top_left = (text_x - padding_x, text_y - text_height - padding_y)
        bottom_right = (text_x + text_width + padding_x, text_y + padding_y)
        cv2.rectangle(frame, top_left, bottom_right, bg_color, -1)
        
        # Draw the black text
        cv2.putText(frame, text, (text_x, text_y), font, font_scale, text_color, thickness, cv2.LINE_AA)
